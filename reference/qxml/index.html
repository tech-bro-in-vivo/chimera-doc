# QXML

QXML 是快应用的标签语言，和 HTML 很类似。 QXML 结合基础组件、事件系统，可以构建出页面的结构。

下面简单展示一下 QXML 的用法：

# 数据绑定

<!--QXML-->
<view> {{message}} </view>
1
2
// page.js
Page({
  data: {
    message: 'Hello QuickApp!'
  }
})

# 列表渲染

<!--QXML-->
<view qa:for="{{array}}"> {{item}} </view>
1
2
// page.js

  data: {
    array: [1, 2, 3, 4, 5]

# 条件渲染

<!--QXML-->
<view qa:if="{{view == 'WEBVIEW'}}"> WEBVIEW </view>
<view qa:elif="{{view == 'APP'}}"> APP </view>
<view qa:else="{{view == 'QuickApp'}}"> QuickApp </view>

# 模板

<!--QXML-->
<template name="staffName">
  <view>
    FirstName: {{firstName}}, LastName: {{lastName}}
  </view>
</template>

<template is="staffName" data="{{...staffA}}"></template>
<template is="staffName" data="{{...staffB}}"></template>
